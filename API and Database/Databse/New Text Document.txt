CREATE PROCEDURE [dbo].[AddCart]
    @CartItems CartModelTVP READONLY, -- TVP containing the list of products
    @Result NVARCHAR(MAX) OUT
AS
BEGIN
    DECLARE @Price DECIMAL(10, 2) -- Set the decimal places and digits before the decimal point

    -- Create a temporary table to hold the data from the TVP
    CREATE TABLE #TempCart
    (
        Id UNIQUEIDENTIFIER,
        IdUser UNIQUEIDENTIFIER,
        IdProduct UNIQUEIDENTIFIER,
        Number INT
    )

    -- Copy data from the TVP into the temporary table
    INSERT INTO #TempCart (Id, IdUser, IdProduct, Number)
    SELECT Id, IdUser, IdProduct, Number FROM @CartItems

    -- Loop through the temporary table
    WHILE EXISTS (SELECT 1 FROM #TempCart)
    BEGIN
        DECLARE @Id UNIQUEIDENTIFIER, @IdUser UNIQUEIDENTIFIER, @IdProduct UNIQUEIDENTIFIER, @Number INT
        SELECT TOP 1 @Id = Id, @IdUser = IdUser, @IdProduct = IdProduct, @Number = Number FROM #TempCart

        DECLARE @check INT = (SELECT COUNT(Id) FROM Cart WHERE IdProduct = @IdProduct)

        -- Check if the product exists in the Product_Detail table
        IF NOT EXISTS (SELECT 1 FROM dbo.Product_Detail WHERE Id = @IdProduct)
        BEGIN
            SET @Result = N'Sản phẩm không tồn tại';
            RETURN;
        END

        -- If the product does not exist in the cart, insert it
        IF @check = 0
        BEGIN
            -- Get the price of the product from the Product table
            SET @Price = (SELECT p.Price FROM dbo.Product_Detail pd JOIN dbo.Product p ON pd.IdProduct = p.Id  WHERE pd.Id = @IdProduct)

            INSERT INTO dbo.Cart
            (
                Id,
                IdUser,
                IdProduct,
                Number,
                TotalPrice
            )
            VALUES
            (
                @Id, -- Id - uniqueidentifier
                @IdUser, -- IdUser - uniqueidentifier
                @IdProduct, -- IdProduct - uniqueidentifier
                @Number, -- Number - int
                (@Price * @Number)  -- TotalPrice - float
            )
            SET @Result = N'Thêm vào giỏ hàng thành công';
        END
        ELSE -- If the product already exists in the cart, update the quantity and total price
        BEGIN
            -- Get the price of the product from the Product table
            SET @Price =  (SELECT p.Price FROM dbo.Product_Detail pd JOIN dbo.Product p ON pd.IdProduct = p.Id  WHERE pd.Id = @IdProduct)

            DECLARE @NumberUpdate INT = ((SELECT Number FROM Cart WHERE IdProduct = @IdProduct) + @Number)
            UPDATE dbo.Cart SET Number = @NumberUpdate, TotalPrice = (@NumberUpdate * @Price) WHERE IdProduct = @IdProduct
            SET @Result = N'Thêm vào giỏ hàng thành công';
        END

        -- Delete the processed product from the temporary table
        DELETE FROM #TempCart WHERE Id = @Id
    END

    -- Drop the temporary table
    DROP TABLE #TempCart
END

CREATE PROCEDURE [dbo].[AddProduct]
    @Id UNIQUEIDENTIFIER,
    @Name NVARCHAR(MAX),
    @IdUser UNIQUEIDENTIFIER,
	@IdProducer UNIQUEIDENTIFIER,
    @IdProductType UNIQUEIDENTIFIER,
    @Material NVARCHAR(MAX),
    @Price FLOAT,
    @Description NVARCHAR(MAX),
    @IdDiscount UNIQUEIDENTIFIER = NULL,
    @CreateBy UNIQUEIDENTIFIER,
    @CreateDate DATE,
    @ModifiedDate DATE,
    @Images NVARCHAR(MAX), -- JSON string containing image data
    @ProductDetails NVARCHAR(MAX), -- JSON string containing product details
    @Result NVARCHAR(MAX) OUT
AS
BEGIN
    SET NOCOUNT ON;

    -- Variables
    DECLARE @IdShop UNIQUEIDENTIFIER;
    DECLARE @cnt INT;
    DECLARE @NameBrand NVARCHAR(MAX);
    DECLARE @NewIdProducer UNIQUEIDENTIFIER = NEWID();
    DECLARE @IdProducerToInsert UNIQUEIDENTIFIER;

    -- Get Shop ID based on User ID
    SELECT @IdShop = Id FROM dbo.Shop WHERE IdUser = @IdUser;

    -- Check if a product with the same name already exists
    SELECT @cnt = COUNT(Id) FROM dbo.Product WHERE Name = @Name;

    -- Get the Shop's Brand Name
    SELECT @NameBrand = BrandName FROM Shop WHERE Id = @IdShop;

    -- Determine whether to insert the existing IdProducer or create a new one
    IF (@NameBrand = N'BlueBird')
    BEGIN
        SET @IdProducerToInsert = @IdProducer; -- Use the provided IdProducer
    END
    ELSE
    BEGIN
        -- Create a new producer if not "BlueBird"
        BEGIN TRY
            -- Start a Producer transaction
            BEGIN TRAN ProducerTransaction;

            DECLARE @FullName NVARCHAR(MAX);
            DECLARE @UserAddress NVARCHAR(MAX);

            -- Get user's full name and address
            SELECT @FullName = FullName, @UserAddress = Address FROM Users WHERE Id = @IdUser;

            -- Insert into Producer table
            INSERT INTO dbo.Producer (Id, Name, Address, UsedStated, CreateBy, CreateDate)
            VALUES (@NewIdProducer, @FullName, @UserAddress, 0, @IdUser, GETDATE());

            SET @IdProducerToInsert = @NewIdProducer; -- Use the newly created IdProducer

            COMMIT TRAN ProducerTransaction;
        END TRY
        BEGIN CATCH
            ROLLBACK TRAN ProducerTransaction;
            SET @Result = N'Lỗi khi thêm Producer'; -- Set error message when an error occurs during Producer insertion
            RETURN;
        END CATCH
    END;

    -- Insert data into the Product table
    BEGIN
        BEGIN TRAN;

        INSERT INTO dbo.Product
        (
            Id,
            Name,
            IdProducer,
            IdProductType,
            Material,
            Price,
            Description,
            CreateBy,
            CreateDate,
            ModifiedDate,
            UsedStatus,
            IdDiscount,
            IdShop
        )
        VALUES
        (
            @Id,
            @Name,
            @IdProducerToInsert,
            @IdProductType,
            @Material,
            @Price,
            @Description,
            @CreateBy,
            @CreateDate,
            @ModifiedDate,
            0,
            @IdDiscount,
            @IdShop
        );

        -- Check for errors during Product insertion
        IF @@ERROR <> 0
        BEGIN
            ROLLBACK TRAN;
            SET @Result = N'Lỗi khi thêm sản phẩm';
        END
        ELSE
        BEGIN
            -- If Product insertion is successful, proceed to insert data into Image and Product_Detail tables

            -- Insert data into the Image table from the JSON string @Images
            INSERT INTO dbo.Image (Id, IdProduct, Img, UsedStatus)
            SELECT NEWID(), @Id, img, UsedStatus
            FROM OPENJSON(@Images)
            WITH (img NVARCHAR(MAX) '$.img', UsedStatus INT '$.UsedStatus');

            -- Insert data into the Product_Detail table from the JSON string @ProductDetails
            INSERT INTO dbo.Product_Detail
            (
                Id,
                IdProduct,
                Number,
                Description,
                CreateBy,
                CreateDate,
                ModifiedDate,
                ModifiedBy,
                Color,
                Size
            )
            SELECT NEWID(), @Id, Number, Description, @CreateBy, GETUTCDATE(), @ModifiedDate, @CreateBy, Color, Size
            FROM OPENJSON(@ProductDetails)
            WITH (Number INT '$.Number', Description NVARCHAR(MAX) '$.Description', Color NVARCHAR(MAX) '$.Color', Size NVARCHAR(MAX) '$.Size');
        END;
    END;

    -- Check for errors during Image and Product_Detail insertion
    IF @@ERROR <> 0
    BEGIN
        ROLLBACK TRAN;
        SET @Result = N'Lỗi khi thêm ảnh hoặc chi tiết sản phẩm';
    END
    ELSE
    BEGIN
        COMMIT TRAN;
        SET @Result = N'Thêm sản phẩm thành công';
    END;
END;

CREATE PROCEDURE [dbo].[FindProduct]
	@search NVARCHAR(MAX),
    @pageIndex INT,
    @pageSize INT
AS
BEGIN
    DECLARE @totalRecords INT;

    -- Tính tổng số bản ghi
    SELECT @totalRecords = COUNT(*) FROM dbo.Product
    WHERE (Name LIKE '%' + @search + '%' 
        OR Description LIKE '%' + @search + '%' 
        OR Material LIKE '%' + @search + '%' 
        OR CONVERT(NVARCHAR(255), CAST(Price AS DECIMAL(38, 0))) LIKE '%' + @search + '%');

    -- Tính vị trí bắt đầu của trang
    DECLARE @startRow INT = (@pageIndex - 1) * @pageSize + 1;
    -- Tính vị trí kết thúc của trang
    DECLARE @endRow INT = @startRow + @pageSize - 1;

    -- Lấy dữ liệu với phân trang
    WITH ProductCTE AS
    (
        SELECT
            p.Id,
            p.Name,
            p.Price,
            p.IdProducer,
            p.IdProductType,
            p.Material,
            p.Description,
            p.IdDiscount, 
            (
                SELECT i.Img
                FROM dbo.Image i
                WHERE i.IdProduct = p.Id
                FOR JSON PATH
            ) AS img,
            (
                SELECT pd.Color, pd.Size, pd.Number, pd.Description
                FROM dbo.Product_Detail pd
                WHERE pd.IdProduct = p.Id
                FOR JSON PATH
            ) AS Product_Detail,
            ROW_NUMBER() OVER (ORDER BY p.Id) AS RowNum
        FROM dbo.Product p
        WHERE EXISTS (
            SELECT 1
            FROM dbo.Image i
            WHERE i.IdProduct = p.Id
        ) OR EXISTS (
            SELECT 1
            FROM dbo.Product_Detail pd
            WHERE pd.IdProduct = p.Id
        )
    )

    -- Trả về kết quả phân trang
    SELECT 
        Id,
        Name,
        Price,
        IdProducer,
        IdProductType,
        Material,
        Description,
        IdDiscount,
        img,
        Product_Detail,
		@totalRecords AS TotalRecord
    FROM ProductCTE
    WHERE RowNum BETWEEN @startRow AND @endRow 
        AND (Name LIKE '%' + @search + '%' 
            OR Description LIKE '%' + @search + '%' 
            OR Material LIKE '%' + @search + '%' 
            OR CONVERT(NVARCHAR(255), CAST(Price AS DECIMAL(38, 0))) LIKE '%' + @search + '%');
END;

IF @search = '' OR @search = NULL
	begin

		JOIN dbo.Image img ON img.IdProduct = p.Id

CREATE PROCEDURE [dbo].[GetAllProduct] 
    @pageIndex INT,
    @pageSize INT,
	@productType Nvarchar(255)
AS
BEGIN
    DECLARE @totalRecords INT;

    -- Tính tổng số bản ghi
    SELECT @totalRecords = COUNT(*) FROM dbo.Product;

    -- Tính vị trí bắt đầu của trang
    DECLARE @startRow INT = (@pageIndex - 1) * @pageSize + 1;
    -- Tính vị trí kết thúc của trang
    DECLARE @endRow INT = @startRow + @pageSize - 1;

    -- Lấy dữ liệu với phân trang
    WITH ProductCTE AS
    (
        SELECT
            p.Id,
            p.Name,
            p.Price,
            p.IdProducer,
            p.IdProductType,
            p.Material,
            p.Description,
            p.IdDiscount, 
            (
                SELECT i.Img
                FROM dbo.Image i
                WHERE i.IdProduct = p.Id
                FOR JSON PATH
            ) AS img,
            (
                SELECT pd.Color, pd.Size, pd.Number, pd.Description, pd.Id, pd.IdProduct
                FROM dbo.Product_Detail pd
                WHERE pd.IdProduct = p.Id
                FOR JSON PATH
            ) AS Product_Detail,
            ROW_NUMBER() OVER (ORDER BY p.Id) AS RowNum
        FROM dbo.Product p
		join dbo.ProductType pt on p.IdProductType = pt.Id
        WHERE( EXISTS (
            SELECT 1
            FROM dbo.Image i
            WHERE i.IdProduct = p.Id
        ) OR EXISTS (
            SELECT 1
            FROM dbo.Product_Detail pd
            WHERE pd.IdProduct = p.Id
        )) and pt.Name = @productType
    )

    -- Trả về kết quả phân trang
    SELECT 
        Id,
        Name,
        Price,
        IdProducer,
        IdProductType,
        Material,
        Description,
        IdDiscount,
        img,
        Product_Detail,
		@totalRecords AS TotalRecord
    FROM ProductCTE
    WHERE RowNum BETWEEN @startRow AND @endRow;
END;

	@ProductDetails NVARCHAR(MAX), -- Chuỗi JSON chứa danh sách chi tiết sản phẩm

BEGIN

		SET 
			Number = j.Number,
			Description = j.Description,
			Color = j.Color,
			Size = j.Size
		FROM dbo.Product_Detail pd
		INNER JOIN OPENJSON(@ProductDetails) WITH (Number INT '$.Number', Description NVARCHAR(MAX) '$.Description', Color NVARCHAR(MAX) '$.Color', Size NVARCHAR(MAX) '$.Size') j 
		ON pd.IdProduct = @Id
		WHERE pd.IdProduct = @Id;

		COMMIT TRAN;
		SET @Result = N'Sửa sản phẩm thành công'; -- Đặt thông báo thành công khi tất cả các bước đều thành công
	END
END
GO
